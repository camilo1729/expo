    class CmdCtrl::ParallelRunner

    This class manages the execution of several commands in parallel. It is highly configurable.

    Attributes

    commands[RW]

    The commands to run (an Array). Shouldn’t be modified after ‘run’ is called.

    results[R]

    The result of the execution. Should only be read after the execution finished. It is an hash of ParallelRunnerResult indexed by CommandBufferer

    timeout[RW]

    Timeout: we consider a command to have failed after that many seconds

    window[RW]

    Window: max number of concurrent processes running (nil => unlimited)

def ParallelRunner::rungroup(cmds, window = nil)
  pr = ParallelRunner::new
  pr.window = window
  cmds.each do |c|
    pr.commands << CommandBufferer::new(Command::new(c))
  end
  tstart = Time::new
  pr.run
  pr.wait
  tstop = Time::new
  pr.group_results.sort.each do |rg|
    puts rg
  end
  puts sprintf("--- #{pr.commands.length} commands executed in %.2f seconds ---", (tstop - tstart))
end

def group_results
  raise "No results yet" if @results.empty? or not finished?
  rgs = []
  @results.each_pair do |c, r|
    ok = false
    rgs.each do |rg|
      if rg.same_as(r.stdout, r.stderr, r.status, r.timeout)
        rg.commands << c
        ok = true
        break
      end
    end
    if !ok
      rgs << ParallelRunnerResultGroup::new(r.stdout, r.stderr, r.status, c, r.timeout)
    end
  end
  return rgs
end
