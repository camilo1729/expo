def duration
  min=Time::now()
  max=Time.local(1986, 7, 17)
  self.each do |t|
    min=t['start_time'] if t['start_time']<min
    max=t['end_time'] if t['end_time']>max
  end
  return max-min
end

def mean_duration
  sum = 0
  time = 0
  self.each { |t| sum += t.duration }
  time = sum / self.length if self.length > 0
  return time
end
